# Run following commands.
1. cd py
2. python main.py
3. cd ../ja
4. mvn clean install
5. java -jar /LuceneFileIndexer/target/LuceneFileIndexer-1.0-SNAPSHOT.jar
6. cd ..
7. cd ui/fsindex-ui
8. npm install
9. npm start
10. Open link http://localhost:3000
11. Click on `Reindex`
